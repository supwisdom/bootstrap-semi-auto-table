  $(function () {

    $table.semiAutoTable({
      pageOption: false,
      menus: [
        {
          title: 'Simple',

          btnClass: 'btn-danger',
          tooltip: 'Tip Tip',

        }
      ],
      useDataTable: false
    })



  });
</script>

<script>
  $(function () {

    var $table = $("#table5");

    var showSelectedRows = function () {
      alert('Selected rows: ' + JSON.stringify($table.semiAutoTable('getSelectedRows')));
    }

    $table.semiAutoTable({
      pageOption: false,
      menus: [
        {
          id: 1,
          title: 'Simple',
          callback: showSelectedRows,
          tag: '20'
        },
        {
          title: 'With Icon',
          icon: 'fa fa-calendar',
          callback: showSelectedRows
        },
        {
          id: 3,
          title: 'Styled',
          btnClass: 'btn-success',
          callback: showSelectedRows
        }
      ],
      useDataTable: false
    })

    $table.semiAutoTable('renderButtonTag', [{id:3, tag: '11'}]);

  });
</script>
